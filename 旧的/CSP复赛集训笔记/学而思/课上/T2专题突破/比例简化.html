2014年第2题：比例简化

计算机最慢的计算是除法和取模运算

要求 ab≥AB；交叉相乘，转化为 aB≥Ab

#include<iostream>
#include<cstdio>
using namespace std;
int gcd(int a, int b)
{
	return b ? gcd(b, a % b) : a;
}
int main()
{
	int a, b, l;
	scanf("%d %d %d", &a, &b, &l);
	double num = (double)a / b;
	int ans_a = -1, ans_b = -1;
	double min = 1e9;
	for(int i = 0 ; i <= l ; ++i)
		for(int j = 1 ; j <= l ; ++j)
			if(gcd(i, j) == 1)
			{
				double nt = (double)i / j;
				double c = nt - num;
				if((c >= 0) && (c < min))
					ans_a = i, ans_b = j, min = c;
			}
	printf("%d %d", ans_a, ans_b);
	return 0;
}
