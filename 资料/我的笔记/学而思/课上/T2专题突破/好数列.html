哈希表（unordered_map）计数

#include<cstdio>
#include<iostream>
#include<algorithm>
#include<unordered_map>
using namespace std;
unordered_map<int,int> mp;
int main()
{
	int n = 0;
	scanf("%d", &n);
	for(int i = 0 ; i < n ; ++i)
	{
		int x = 0;
		scanf("%d", &x);
		mp[x]++;
	}
	int ans = 0;
	for(auto i = mp.begin() ; i != mp.end() ; ++i)
	{
		int number = i -> first;
		int count = i -> second;
		if(count < number)
			ans += count;
		else
			ans += count - number;
	}
	printf("%d", ans);
	return 0;
}
