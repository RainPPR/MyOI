# 组合数

## v1
```cpp
#include <cstdio>

const int N = 2010, MOD = 1e9 + 7;
int c[N][N];

int main()
{
	for (int i = 0 ; i < N ; ++i)
		for (int j = 0 ; j <= i ; ++j)
			if (!j)
				c[i][j] = 1;
			else
				c[i][j] = (c[i - 1][j] + c[i - 1][j - 1]) % MOD;
	int n, a, b;
	scanf("%d", &n);
	while (n--)
	{
		scanf("%d %d", &a, &b);
		printf("%d\n", c[a][b]);
	}
	return 0;
}
```

## v2
```cpp
#include <cstdio>

typedef long long ll;
const int N = 100010, MOD = 1e9 + 7;

int fact[N], infact[N];

int qmi(int a, int k, int p)
{
	int res = 1;
	while (k)
	{
		if (k & 1)
			res = (ll)res * a % p;
		a = (ll)a * a % p;
		k >>= 1;
	}
	return res;
}

int main()
{
	fact[0] = infact[0] = 1;
	for (int i = 1 ; i < N ; ++i)
	{
		fact[i] = (ll)fact[i - 1] * i % MOD;
		infact[i] = (ll)infact[i - 1] * qmi(i, MOD - 2, MOD) % MOD;
	}
	int n, a, b;
	scanf("%d", &n);
	while (n--)
	{
		scanf("%d %d", &a, &b);
		printf("%lld\n", (ll)fact[a] * infact[b] % MOD * infact[a - b] % MOD);
	}
	return 0;
}
```
